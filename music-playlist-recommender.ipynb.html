164  ### 5.2.1 Getting song audio features 
165  #### This function is to pull the audio features for a given track needed to eventually be used to recommend songs 
166  #%% 
167  def get_song(artist_name,name, year):
168      song_info = defaultdict()
169      results = sp.search(q= 'track: {} artists.str.contains({})'.format(name,year), limit=1)
170      if results['tracks']['items'] == []:
171          return None
172  
173      results = results['tracks']['items'][0]
174      track_id = results['id']
175      audio_features = sp.audio_features(track_id)[0]
176      
177      song_info['artist_name'] = [artist_name]
178      song_info['track_name'] = [track_name]
179      song_info['year'] = [year]
180      song_info['duration_ms'] = [results['duration_ms']]
181      song_info['popularity'] = [results['popularity']]
182  
183      for key, value in audio_features.items():
184          song_info[key] = value
185  
186      return pd.DataFrame(song_info)
187  
188  #%% md 
189  ### 5.2.2 Getting song audio information 
190  #### This function is to pull the information such as track name, year, and artist name for a given songÂ¶ 
191  #%% 
192  def get_song_info(song, spotify_data):
193      
194      try:
195          song_info = spotify_data[(spotify_data['artist_name'] == song['artist_name'])&(spotify_data['track_name'] == song['track_name']) 
196                                  & (spotify_data['year'] == song['year'])].iloc[0]
197          return song_info
198      
199      except IndexError:
200          return get_song(song['artist_name'],song['track_name'], song['year'])
201  #%% md 
202  ### 5.2.3 Generating the median for each audio feature 
203  #### In order to recommend a list of songs based on a given one, we need to use audio features. The best way to do this is to caculate the median of the value for each feature. 
204  #%% 
205  def get_median_vector(song_list, spotify_data):
206      
207      song_vectors = []
208      
209      for song in song_list:
210          song_info = get_song_info(song, spotify_data)
211          if song_info is None:
212              print('Warning: {} does not exist in Spotify or in database'.format(song['name']))
213              continue
214          song_vector = song_info[audio_features].values
215          song_vectors.append(song_vector)  
216      
217      song_matrix = np.array(list(song_vectors))
218      return np.median(song_matrix, axis=0)
219  #%% md 
220  ### 5.2.4 Creating a dictionary that will be our input in the final model 
221  #%% 
222  def flatten_dict(dict_list):
223      
224      flattened_dict = defaultdict()
225      for key in dict_list[0].keys():
226          flattened_dict[key] = []
227      
228      for dictionary in dict_list:
229          for key, value in dictionary.items():
230              flattened_dict[key].append(value)
231      return flattened_dict
232  #%% md 
233  ## 6.2 Building recommender model 
234  #%% 
235  def song_recommender(song_list, spotify_data,n_songs = 15):
236      
237      metadata_cols = ['artist_name','track_name','year']
238      song_dict = flatten_dict(song_list)
239      
240      song_mean = get_median_vector(song_list, spotify_data)
241      scaler = song_cluster_pipeline.steps[0][1]
242      scaled_data = scaler.transform(spotify_data[audio_features])
243      scaled_song_center = scaler.transform(song_mean.reshape(1, -1))
244      distances = cdist(scaled_song_center, scaled_data, 'cosine')
245      index = list(np.argsort(distances)[:, :n_songs][0])
246      
247      rec_songs = spotify_data.iloc[index]
248      rec_songs = rec_songs[~rec_songs['track_name'].isin(song_dict['track_name'])]
249      return rec_songs[metadata_cols].reset_index(drop=True)
